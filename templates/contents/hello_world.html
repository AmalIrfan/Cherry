The following is a program written in flat assembly, for amd64 linux 

format ELF64 executable
segment readable executable
entry _start

_start:
    mov rax, 1
    mov rdi, 1
    mov rsi, msg
    mov rdx, msg_len
    syscall

    mov rax, 60
    mov rdi, 0
    syscall

segment readable writable

msg:
    db "Hello, World!", 10

msg_len = $ - msg
